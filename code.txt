#Build docker file into container
docker build -t demoapp3 .

# List images
docker images

#Run container
docker run -d -p 8080:80 --name demoapp3 demoapp3:latest

#List running containers

docker ps 

#List All containers inc stopped

docker ps -a 

#Remove container

docker rm demoapp3

#Remove images
docker rmi [container id]
#ECR command to tag 
docker tag demoapp3:latest  [ECR URL]:latest

#ECR command to push to ECR
docker push [ecr url]:latest
